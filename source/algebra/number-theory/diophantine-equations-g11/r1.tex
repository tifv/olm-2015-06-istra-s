\section*{Уравнения}

% $authors:
% - Андрей Меньщиков

Во всех задачах нужно решить уравнение в натуральных числах.

\begingroup \multicolsep=\parskip
\begin{multicols}{2}
\begin{problems}

\item
$a^3 + 1 = 3^n$.

\item
$m! + 12 = n^2$.

\item
$3 \cdot 2^x + 1 = y^2$.

\item
$x^3 + 13 = 2^y$.

\item
$x^4 - 2 y^2 = 1$.

\item
$|2^m - 3^n| = 1$.

\item
$x^3 + 7 = y^2$.

\item
$4 x y - x - y = z^2$.

\item
$1 + 2^x + 2^{2x+1} = y^2$.

\item
$n^5 + n^4 = 7^m - 1$.

\item
$x^2 + 5 = y^3$.

\par\strut % even the columns

\end{problems}
\end{multicols}
\endgroup
